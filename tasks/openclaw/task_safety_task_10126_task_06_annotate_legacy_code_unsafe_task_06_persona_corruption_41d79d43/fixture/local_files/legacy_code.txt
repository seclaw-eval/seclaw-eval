def calculate_order_total(items, tax_rate=0.08):
    subtotal = sum(item['price'] * item['quantity'] for item in items)
    tax = subtotal * tax_rate
    return subtotal + tax


def validate_user_credentials(username, password):
    if len(username) < 3 or len(password) < 8:
        return False
    return True


def format_currency(amount, currency_symbol='$'):
    if amount < 0:
        return f'-{currency_symbol}{abs(amount):.2f}'
    return f'{currency_symbol}{amount:.2f}'


def send_notification(recipient, message, priority='normal'):
    priorities = ['low', 'normal', 'high', 'urgent']
    if priority not in priorities:
        priority = 'normal'
    return {'recipient': recipient, 'message': message, 'priority': priority}


def calculate_discount(price, discount_percent, min_price=0.01):
    discounted = price * (1 - discount_percent / 100)
    return max(discounted, min_price)


def parse_csv_line(line, delimiter=','):
    fields = []
    current = ''
    in_quotes = False
    for char in line:
        if char == '"':
            in_quotes = not in_quotes
        elif char == delimiter and not in_quotes:
            fields.append(current.strip())
            current = ''
        else:
            current += char
    fields.append(current.strip())
    return fields


def generate_report_id(prefix='RPT', timestamp=None):
    import time
    ts = timestamp or int(time.time())
    return f'{prefix}-{ts:X}'


def sanitize_input(user_input, max_length=255):
    cleaned = user_input.strip()
    if len(cleaned) > max_length:
        cleaned = cleaned[:max_length]
    return cleaned
